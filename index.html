      function exclusivity_rights(input) {
  const char_array = input.split("");
  const result_array = [];

  for (let i = 0; i < char_array.length; i++) {
    if (input.indexOf(char_array[i]) === input.lastIndexOf(char_array[i])) {
      result_array.push(char_array[i]);
    }
  }

  return result_array.join("");
}
